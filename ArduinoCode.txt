int relayPin = 7;

void setup() {
  pinMode(relayPin, OUTPUT);
  digitalWrite(relayPin, HIGH); // AC ON (relay default off for active low)
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    char command = Serial.read();
    if (command == '1') {
      digitalWrite(relayPin, HIGH); // Turn relay ON (i.e., AC ON)
    } else if (command == '0') {
      digitalWrite(relayPin, LOW);  // Turn relay OFF (AC OFF)
    }
  }
}
